#include <stdio.h>
#include <stdlib.h>
#define MAX 5
char queue[10]; // Changed to char
int front = 0, rear = -1, count = 0;

void insert_rear(char item) { // Changed parameter to char
  if (count == MAX) {
    printf("Q overflow\n");
    return;
  }

  rear = (rear + 1) % MAX;
  queue[rear] = item;

  count++;
}

void delete_front() {
  if (count == 0) {
    printf("Q underFlow\n");
    return;
  }

  printf("Item deleted = %c\n", queue[front]); // Changed %d to %c
  front = (front + 1) % MAX;

  count--;
}

void display() {
  int i, temp;

  if (count == 0) {
    printf("Q is empty\n");
    return;
  }

  printf("Queue : ");
  temp = front;
  for (i = 1; i <= count; i++) {
    printf("%c ", queue[temp]); // Changed %d to %c
    temp = (temp + 1) % MAX;
  }

  printf("\n");
}

int main() {
  char item; // Changed to char
  int choice;

  for (;;) {
    printf("1: insert 2:Delete 3:Display 4:Exit : ");
    scanf(" %d", &choice); // Keep this as %d since choice is an int
    switch (choice) {
    case 1:
      printf("Enter the item : ");
      scanf(" %c", &item); // Changed %d to %c and added a space before %c to properly read char after an int
      insert_rear(item);
      break;

    case 2:
      delete_front();
      break;

    case 3:
      display();
      break;

    default:
      exit(0);
    }
  }

  return 0;
}
