//* SWAPPING OF TWO NUMBERS WITHOUT USING
THIRD VARIABLE *//

void main()

{

a = a + b;

//* FOBNOCCI SERIES *//
#include <stdio.h>

void main()

{

int n, first = 0, second = 1, next, i;

printf("Enter the number of terms: ");
scanf("%d", &n);

printf("Fibonacci Series: ");
for (i = 0; i < n; i++) {

if (i <= 1)

next = i;
else {

next = first + second;
first = second;
second = next;

}

printf("%d ", next);

}

}

OUTPUT :

Enter the number of terms: 5
Fibonacci Series: 0 1 1 2 3


//* TO CHECK WETHER A NUMBER IS PRIME OR NOT

*//

#include <stdio.h>
void main()

{

int num, i;

printf("Enter a number: ");
scanf("%d", &num);

if (num <= 1)

{

printf("%d is not a prime number.\n", num);
return;

}

for (i = 2; i <= num / 2; i++)

{

if (num % i == 0)

{

printf("%d is not a prime number.\n", num);
return;

}

}

printf("%d is a prime number.\n", num);

}

//* TO CHECK WETHER A NUMBER IS PALINDROME

OR NOT *//
#include <stdio.h>

void main()

{

int num, rev = 0, temp;

printf("Enter a number: ");
scanf("%d", &num);

temp = num;
while (temp != 0)

{

rev = rev * 10 + temp % 10;
temp /= 10;

}

printf("%s\n", (num == rev) ? "Palindrome" : "Not
Palindrome");

}
OUTPUT :

Enter a number: 121
Palindrome

Enter a number: 110
Not Palindrome

//* ARMSTRONG NUMBER *//

#include <stdio.h>
void main()

{

int num, sum = 0, temp, digit;

printf("Enter a positive integer: ");
scanf("%d", &num);

for (temp = num; temp != 0; temp /= 10)

{

digit = temp % 10;

sum += digit * digit * digit;

}

if (sum == num)

printf("%d is an Armstrong number.\n", num);


else

}

printf("%d is not an Armstrong number.\n", num);

OUTPUT :

Enter a positive integer: 153
153 is an Armstrong number.

Enter a positive integer: 123

123 is not an Armstrong number.

//* REVERSING A NUMBER *//

#include <stdio.h>
void main()

{

int num, reverse = 0;

printf("Enter an integer: ");
scanf("%d", &num);

while (num != 0)

{

reverse = reverse * 10 + num % 10;
num /= 10;

}

printf("Reversed number: %d\n", reverse);

}

//* PRINT HELLO WORLD WITHOUT USING
SEMICOLON *//

#include <stdio.h>
#define SEMICOLON ;
int main()

{

if (printf("Hello World!\n")) {}

return 0;

}

OUPUT :

Hello World!
NOTE :

1. We use a preprocessor directive #define to define SEMICOLON as ;.

2. Then, in the main() function, we use an if statement with
printf("Hello, World!\n") as the condition. printf() returns the
number     of characters printed, which in this case will be the number
of characters in "Hello, World!\n" (including the newline character).
Since this value will be non-zero, the condition evaluates to true, and
the empty block {} gets executed.

3.  Although this code technically does not have a semicolon after the
printf() statement, it relies on the semicolon present in the printf()
function itself.

//* LEAP YEAR *//
#include <stdio.h>

void main()

{

int year;

printf("Enter a year: ");
scanf("%d", &year);

if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0))

printf("%d is a leap year.\n", year);


else

}

printf("%d is not a leap year.\n", year);

OUTPUT :

Enter a year: 2005
2005 is not a leap year.

Enter a year: 2004
2004 is a leap year.
